016    package org.codehaus.groovy.grails.plugins.searchable.compass.mapping;
017    
018    import org.codehaus.groovy.grails.commons.GrailsDomainClass;
019    
020    import java.util.Collection;
021    import java.util.List;
022    
023    /**
024     * Maps a searchable GrailsDomainClass to Compass
025     * 
026     * @author Maurice Nicholson
027     */
028    public interface SearchableGrailsDomainClassCompassClassMapper {
029    
030        /**
031         * Get the property mappings for the given GrailsDomainClass
032         * @param grailsDomainClass the Grails domain class
033         * @param searchableGrailsDomainClasses a collection of searchable GrailsDomainClass instances
034         * @param searchableValue the searchable value: true|false|Map|Closure
035         * @param excludedProperties a List of properties NOT to map; may be ignored by impl
036         * @return a List of CompassClassPropertyMapping
037         */
038        List getCompassClassPropertyMappings(GrailsDomainClass grailsDomainClass, Collection searchableGrailsDomainClasses, Object searchableValue, List excludedProperties);
039    
040        /**
041         * Get the CompassClassMapping for the given GrailsDomainClass
042         * 
043         * @param grailsDomainClass the Grails domain class
044         * @param searchableGrailsDomainClasses a collection of searchable GrailsDomainClass instances
045         * @return the CompassClassMapping
046         */
047        CompassClassMapping getCompassClassMapping(GrailsDomainClass grailsDomainClass, Collection searchableGrailsDomainClasses);
048    
049        /**
050         * Get the CompassClassMapping  for the given GrailsDomainClass and "searchable" value
051         * @param grailsDomainClass the Grails domain class
052         * @param searchableGrailsDomainClasses a collection of searchable GrailsDomainClass instances
053         * @param searchableValue the searchable value: true|false|Map|Closure
054         * @param excludedProperties a List of properties NOT to map; may be ignored by impl
055         * @return the CompassClassMapping
056         */
057        CompassClassMapping getCompassClassMapping(GrailsDomainClass grailsDomainClass, Collection searchableGrailsDomainClasses, Object searchableValue, List excludedProperties);
058    
059        /**
060         * Does the implementation handle a "searchable" value of the given type?
061         * @param searchableValue the searchable value
062         * @return true if this implementation understands the type
063         */
064        boolean handlesSearchableValue(Object searchableValue);
065    }
