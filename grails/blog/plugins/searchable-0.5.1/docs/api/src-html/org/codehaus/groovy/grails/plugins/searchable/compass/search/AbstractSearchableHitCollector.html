016    package org.codehaus.groovy.grails.plugins.searchable.compass.search;
017    
018    import org.apache.commons.collections.MapUtils;
019    import org.codehaus.groovy.runtime.InvokerHelper;
020    import org.compass.core.CompassHits;
021    
022    import java.util.Map;
023    
024    /**
025     * @author Maurice Nicholson
026     */
027    public abstract class AbstractSearchableHitCollector implements SearchableHitCollector {
028    
029        /**
030         * Collect hits and return something; maybe a list of results, maybe a single result, ...
031         * @param hits compass hits
032         * @param options
033         * @return the result object
034         */
035        public Object collect(CompassHits hits, Map options) {
036            boolean reload = MapUtils.getBooleanValue(options, "reload");
037            return collect(hits, reload, options);
038        }
039    
040        abstract Object collect(CompassHits hits, boolean reload, Map options);
041    
042        protected Object getObject(Object data, boolean reload) {
043            if (!reload) return data;
044            Object id = InvokerHelper.getProperty(data, "id");
045            return InvokerHelper.invokeStaticMethod(data.getClass(), "get", new Object[] {id});
046        }
047    }
