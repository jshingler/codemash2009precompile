016    package org.codehaus.groovy.grails.plugins.searchable.compass.search;
017    
018    import org.apache.commons.collections.MapUtils;
019    import org.codehaus.groovy.grails.plugins.searchable.SearchableUtils;
020    import org.compass.core.CompassHits;
021    
022    import java.util.ArrayList;
023    import java.util.Collections;
024    import java.util.List;
025    import java.util.Map;
026    
027    /**
028     * Collects hits within a range denoted by offset and max
029     *
030     * @author Maurice Nicholson
031     */
032    public class DefaultSearchableSubsetHitCollector extends AbstractSearchableHitCollector implements SearchableHitCollector {
033    
034        /**
035         * Collect and return the subset of hits with the range offset..offset + max
036         * 
037         * @param hits
038         * @param options
039         * @return
040         */
041        public Object collect(CompassHits hits, boolean reload, Map options) {
042            if (hits.length() == 0) {
043                return Collections.EMPTY_LIST;
044            }
045    
046            int offset = MapUtils.getIntValue(options, "offset");
047            int max = MapUtils.getIntValue(options, "max");
048            List collectedHits = new ArrayList(max);
049            int low = offset;
050            int high = Math.min(low + max, hits.length());
051            while (low < high) {
052                collectedHits.add(getObject(hits.data(low++), reload));
053            }
054    
055            return collectedHits;
056        }
057    }
