016    package org.codehaus.groovy.grails.plugins.searchable.compass;
017    
018    import org.apache.commons.logging.Log;
019    import org.apache.commons.logging.LogFactory;
020    import org.codehaus.groovy.grails.plugins.searchable.util.TimeUtils;
021    import org.compass.gps.CompassGps;
022    import org.springframework.util.Assert;
023    
024    /**
025     * @author Maurice Nicholson
026     */
027    public class CompassGpsUtils {
028        private static final Log LOG = LogFactory.getLog(CompassGpsUtils.class);
029    
030        /**
031         * Calls CompassGps's index method, starting and stopping it if required
032         * @param compassGps aCompassGps instance, cannot be null
033         * @param clazz the Class to index instances of, may be null
034         */
035        public static void index(CompassGps compassGps, Class clazz) {
036            Assert.notNull(compassGps, "compassGps cannot be null");
037    
038            long start = System.currentTimeMillis();
039            LOG.info("Starting Searchable Plugin bulk index");
040            boolean gpsRunning = compassGps.isRunning();
041            try {
042                if (!gpsRunning) {
043                    compassGps.start();
044                }
045                if (clazz != null) {
046                    compassGps.index(new Class[] {clazz});
047                } else {
048                    compassGps.index();
049                }
050            } finally {
051                if (!gpsRunning) {
052                    compassGps.stop();
053                }
054            }
055            LOG.info("Finished Searchable Plugin bulk index, " + TimeUtils.formatMillisAsShortHumanReadablePeriod(System.currentTimeMillis() - start));
056        }
057    }
