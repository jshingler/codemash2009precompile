016    package org.codehaus.groovy.grails.plugins.searchable.compass.search;
017    
018    import org.compass.core.CompassQuery;
019    import org.compass.core.CompassSession;
020    import org.compass.core.Compass;
021    import org.compass.core.CompassQueryBuilder;
022    import org.codehaus.groovy.grails.commons.GrailsApplication;
023    import org.codehaus.groovy.grails.commons.GrailsDomainClass;
024    import org.codehaus.groovy.grails.plugins.searchable.compass.mapping.CompassMappingUtils;
025    import org.codehaus.groovy.grails.plugins.searchable.SearchableUtils;
026    import org.apache.commons.logging.Log;
027    import org.apache.commons.logging.LogFactory;
028    
029    import java.util.*;
030    import java.io.Serializable;
031    import java.lang.reflect.Method;
032    
033    /**
034     * @author Maurice Nicholson
035     */
036    public class MoreLikeThisCompassQueryBuilder extends AbstractSearchableCompassQueryBuilder implements SearchableCompassQueryBuilder {
037        private static final Log LOG = LogFactory.getLog(MoreLikeThisCompassQueryBuilder.class);
038    
039        public MoreLikeThisCompassQueryBuilder(Compass compass) {
040            super(compass);
041        }
042    
043        public CompassQuery buildQuery(GrailsApplication grailsApplication, CompassSession compassSession, Map options, Object args) {
044            List argv = new ArrayList(Arrays.asList((Object[]) args));
045            // discard options Map from args
046            for (Iterator iter = argv.iterator(); iter.hasNext(); ) {
047                if (iter.next() instanceof Map) {
048                    iter.remove();
049                    break;
050                }
051            }
052            Object instance = null;
053            Collection domainClasses = SearchableUtils.getGrailsDomainClasses(grailsApplication);
054            for (Iterator iter = argv.iterator(); iter.hasNext(); ) {
055                Object obj = iter.next();
056                for (Iterator di = domainClasses.iterator(); di.hasNext(); ) {
057                    GrailsDomainClass gdc = (GrailsDomainClass) di.next();
058                    if (obj.getClass().equals(gdc.getClazz())) {
059                        instance = obj;
060                        break;
061                    }
062                }
063            }
064    
065            Serializable id;
066            String alias;
067            if (instance != null) {
068                Class clazz = instance.getClass();
069                alias = CompassMappingUtils.getMappingAlias(getCompass(), clazz);
070                id = SearchableUtils.getIdent(getCompass(), alias, instance);
071            } else {
072                Class clazz = (Class) options.get("class");
073                if (clazz == null && options.containsKey("match")) {
074                    clazz = (Class) options.get("match");
075                }
076                alias = (String) options.get("alias");
077    
078                for (Iterator iter = argv.iterator(); iter.hasNext(); ) {
079                    Object it = iter.next();
080                    if (it instanceof Class) {
081                        clazz = (Class) it;
082                        alias = null; // override alias with class instance
083                        iter.remove();
084                        break;
085                    }
086                }
087                if (alias == null) {
088                    if (clazz != null) {
089                        alias = CompassMappingUtils.getMappingAlias(getCompass(), clazz);
090                    } else {
091                        throw new IllegalArgumentException("Neither class nor alias was provided for MoreLikeThis query builder: please provide one in either options (Map) or arguments (Object[])");
092                    }
093                }
094                id = (Serializable) options.get("id");
095                if (id == null) {
096                    if (!argv.isEmpty()) {
097                        id = (Serializable) argv.get(0);
098                    } else {
099                        throw new IllegalArgumentException("Unable to find id for MoreLikeThis query builder. Please provide one in either options (Map) or arguments (Object[])");
100                    }
101                }
102            }
103            CompassQueryBuilder.CompassMoreLikeThisQuery moreLikeThisQuery = compassSession.queryBuilder().moreLikeThis(alias, id);
104            if (options.size() > 0) {
105                Method[] methods = CompassQueryBuilder.CompassMoreLikeThisQuery.class.getDeclaredMethods();
106                for (int i = 0; i < methods.length; i++) {
107                    Method method = methods[i];
108                    String name = method.getName();
109                    if (!name.startsWith("set") || name.length() < 4) {
110                        continue;
111                    }
112                    if (method.getParameterTypes().length != 1) {
113                        continue;
114                    }
115                    name = name.substring(3, 4).toLowerCase() + (name.length() > 4 ? name.substring(4) : "");
116                    if (!options.containsKey(name)) {
117                        continue;
118                    }
119                    Object value = options.get(name);
120                    // convert List to String[]?
121                    if (method.getParameterTypes()[0].equals(String[].class) && value instanceof List) {
122                        value = ((List) value).toArray(new String[((List) value).size()]);
123                    }
124                    try {
125                        method.invoke(moreLikeThisQuery, new Object[] {value});
126                    } catch (Exception ex) {
127                        LOG.warn("Failed to invoke method " + moreLikeThisQuery.getClass().getName() + "#" + method.getName() + " with value [" + value + "]", ex);
128                    }
129                }
130            }
131    
132            CompassQuery query = moreLikeThisQuery.toQuery();
133            options.remove("class"); // discard class option so it's not added as match alias in CompassQuery by option helper; todo change "class" option to "match: instead
134            options.put("class", options.get("match"));
135            query = applyOptions(grailsApplication, getCompass(), compassSession, query, options);
136    //        System.out.println("Query [" + query.toString() + "]");
137            return query;
138        }
139    }
