016    package org.codehaus.groovy.grails.plugins.searchable.compass.config.mapping;
017    
018    import org.codehaus.groovy.grails.commons.GrailsDomainClass;
019    
020    import java.util.Collection;
021    import java.util.Set;
022    import java.util.HashSet;
023    import java.util.Iterator;
024    
025    /**
026     * @author Maurice Nicholson
027     */
028    public abstract class AbstractSimpleSearchableGrailsDomainClassMappingConfigurator implements SearchableGrailsDomainClassMappingConfigurator {
029    
030        /**
031         * Returns the collection of GrailsDomainClasses that are mapped by this instance
032         * @param grailsDomainClasses ALL domain classes
033         * @return the collection of domain classes mapped by this instance
034         */
035        public Collection getMappedBy(Collection grailsDomainClasses) {
036            Set mappedBy = new HashSet();
037            for (Iterator iter = grailsDomainClasses.iterator(); iter.hasNext(); ) {
038                GrailsDomainClass grailsDomainClass = (GrailsDomainClass) iter.next();
039                if (isMappedBy(grailsDomainClass)) {
040                    mappedBy.add(grailsDomainClass);
041                }
042            }
043            return mappedBy;
044        }
045    
046        /**
047         * Is the given domain class mapped by this instance?
048         * @param grailsDomainClass domain class
049         * @return true if this mapper handles the given domain class
050         */
051        public abstract boolean isMappedBy(GrailsDomainClass grailsDomainClass);
052    }
