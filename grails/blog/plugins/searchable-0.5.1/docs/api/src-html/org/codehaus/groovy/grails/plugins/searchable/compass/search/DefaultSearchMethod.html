033    /**
034     * The default search method implementation
035     *
036     * @author Maurice Nicholson
037     */
038    public class DefaultSearchMethod extends AbstractSearchableMethod implements SearchableMethod {
039        private static Log LOG = LogFactory.getLog(DefaultSearchMethod.class);
040    
041        private GrailsApplication grailsApplication;
042        private SearchableCompassQueryBuilder compassQueryBuilder;
043        private SearchableHitCollector hitCollector;
044        private SearchableSearchResultFactory searchResultFactory;
045    
046        public DefaultSearchMethod(String methodName, Compass compass, GrailsApplication grailsApplication, SearchableMethodFactory methodFactory, Map defaultOptions) {
047            super(methodName, compass, methodFactory, defaultOptions);
048            this.grailsApplication = grailsApplication;
049        }
050    
051        public Object invoke(Object[] args) {
052            Assert.notNull(args, "args cannot be null");
053            Assert.notEmpty(args, "args cannot be empty");
054    
055            SearchableMethod suggestQueryMethod = getMethodFactory().getMethod("suggestQuery");
056    
057            SearchCompassCallback searchCallback = new SearchCompassCallback(getCompass(), getDefaultOptions(), args);
058            searchCallback.setGrailsApplication(grailsApplication);
059            searchCallback.setCompassQueryBuilder(compassQueryBuilder);
060            searchCallback.setHitCollector(hitCollector);
061            searchCallback.setSearchResultFactory(searchResultFactory);
062            searchCallback.setSuggestQueryMethod(suggestQueryMethod);
063            return doInCompass(searchCallback);
064        }
065    
066        public void setCompassQueryBuilder(SearchableCompassQueryBuilder compassQueryBuilder) {
067            this.compassQueryBuilder = compassQueryBuilder;
068        }
069    
070        public void setHitCollector(SearchableHitCollector hitCollector) {
071            this.hitCollector = hitCollector;
072        }
073    
074        public void setSearchResultFactory(SearchableSearchResultFactory searchResultFactory) {
075            this.searchResultFactory = searchResultFactory;
076        }
077    
078        public void setGrailsApplication(GrailsApplication grailsApplication) {
079            this.grailsApplication = grailsApplication;
080        }
081    
082        public static class SearchCompassCallback implements CompassCallback {
083            private Object[] args;
084            private Map defaultOptions;
085            private GrailsApplication grailsApplication;
086            private SearchableCompassQueryBuilder compassQueryBuilder;
087            private SearchableHitCollector hitCollector;
088            private SearchableSearchResultFactory searchResultFactory;
089            private SearchableMethod suggestQueryMethod;
090            private static final String[] OVERRIDE_WITH_DEFAULTS_IF_NULL = {"max", "offset"};
091    
092            public SearchCompassCallback(Compass compass, Map defaultOptions, Object[] args) {
093                this.args = args;
094                this.defaultOptions = defaultOptions;
095            }
096    
097            public Object doInCompass(CompassSession session) throws CompassException {
098                Map options = SearchableMethodUtils.getOptionsArgument(args, defaultOptions, OVERRIDE_WITH_DEFAULTS_IF_NULL);
099                CompassQuery compassQuery = compassQueryBuilder.buildQuery(grailsApplication, session, options, args);
100                long start = System.currentTimeMillis();
101                CompassHits hits = compassQuery.hits();
102                if (LOG.isDebugEnabled()) {
103                    long time = System.currentTimeMillis() - start;
104                    LOG.debug("query: [" + compassQuery + "], [" + hits.length() + "] hits, took [" + time + "] millis");
105                }
106                if (hitCollector == null && searchResultFactory == null) {
107                    Assert.notNull(options.get("result"), "Missing 'result' option for search/query method: this should be provided if hitCollector/searchResultFactory are null to determine the type of result to return");
108                    String result = (String) options.get("result");
109                    if (result.equals("top")) {
110                        hitCollector = new DefaultSearchableTopHitCollector();
111                        searchResultFactory = new SearchableHitsOnlySearchResultFactory();
112                    } else if (result.equals("every")) {
113                        hitCollector = new DefaultSearchableEveryHitCollector();
114                        searchResultFactory = new SearchableHitsOnlySearchResultFactory();
115                    } else if (result.equals("searchResult")) {
116                        hitCollector = new DefaultSearchableSubsetHitCollector();
117                        searchResultFactory = new SearchableSubsetSearchResultFactory();
118                    } else if (result.equals("count")) {
119                        hitCollector = new CountOnlyHitCollector();
120                        searchResultFactory = new SearchableHitsOnlySearchResultFactory();
121                    } else {
122                        throw new IllegalArgumentException("Invalid 'result' option for search/query method [" + result + "]. Supported values are ['searchResult', 'every', 'top']");
123                    }
124                }
125                Object collectedHits = hitCollector.collect(hits, options);
126                Object searchResult = searchResultFactory.buildSearchResult(hits, collectedHits, options);
127    
128                doWithHighlighter(collectedHits, hits, searchResult, options);
129    
130                Object suggestOption = options.get("suggestQuery");
131                if (searchResult instanceof Map && suggestOption != null) {
132                    addSuggestedQuery((Map) searchResult, suggestOption);
133                }
134                return searchResult;
135            }
136    
137            private void addSuggestedQuery(Map searchResult, Object suggestOption) {
138                if (suggestOption instanceof Boolean && suggestOption.equals(Boolean.FALSE)) {
139                    return;
140                }
141                if (suggestOption instanceof String && !Boolean.valueOf((String) suggestOption).booleanValue()) {
142                    return;
143                }
144                Object[] suggestArgs = new Object[args.length];
145                for (int i = 0; i < args.length; i++) {
146                    if (args[i] instanceof Map) {
147                        Map searchOptions = (Map) args[i];
148                        Map suggestOptions = new HashMap(searchOptions);
149                        if (suggestOption instanceof Map) {
150                            suggestOptions.putAll((Map) suggestOption);
151                        }
152                        suggestOptions.remove("suggestQuery"); // remove the option itself
153                        // add other defaults for use from search method
154                        if (!suggestOptions.containsKey("allowSame")) {
155                            suggestOptions.put("allowSame", Boolean.FALSE);
156                        }
157                        suggestArgs[i] = suggestOptions;
158                    } else {
159                        suggestArgs[i] = args[i];
160                    }
161                }
162                searchResult.put("suggestedQuery", suggestQueryMethod.invoke(suggestArgs));
163            }
164    
165            public void doWithHighlighter(Object collectedHits, CompassHits hits, Object searchResult, Map options) {
166                if (!(collectedHits instanceof Collection)) {
167                    return;
168                }
169                Closure withHighlighter = (Closure) options.get("withHighlighter");
170                if (withHighlighter == null) {
171                    return;
172                }
173                withHighlighter = (Closure) withHighlighter.clone();
174                int offset = org.apache.commons.collections.MapUtils.getIntValue(options, "offset");
175                for (int i = 0, length = ((Collection) collectedHits).size(); i < length; i++) {
176                    withHighlighter.call(new Object[] {
177                        hits.highlighter(offset + i), new Integer(i), searchResult
178                    });
179                }
180            }
181    
182            public void setGrailsApplication(GrailsApplication grailsApplication) {
183                this.grailsApplication = grailsApplication;
184            }
185    
186            public void setCompassQueryBuilder(SearchableCompassQueryBuilder compassQueryBuilder) {
187                this.compassQueryBuilder = compassQueryBuilder;
188            }
189    
190            public void setHitCollector(SearchableHitCollector hitCollector) {
191                this.hitCollector = hitCollector;
192            }
193    
194            public void setSearchResultFactory(SearchableSearchResultFactory searchResultFactory) {
195                this.searchResultFactory = searchResultFactory;
196            }
197    
198            public void setSuggestQueryMethod(SearchableMethod suggestQueryMethod) {
199                this.suggestQueryMethod = suggestQueryMethod;
200            }
201        }
