034    /**
035    * A Groovy CompassQuery builder, taking nested closures and dynamic method
036    * invocation in the Groovy-builder style
037    *
038    * Note: Instances of this class are NOT thread-safe: you should create one
039    * for the duration of your CompassSession then discard it
040    *
041    * @author Maurice Nicholson
042    */
043    public class GroovyCompassQueryBuilder extends GroovyObjectSupport {
044        private CompassQueryBuilder queryBuilder;
045    
046        /**
047         * Constructor
048         * @param queryBuilder a CompassQueryBuilder used internally to build the query
049         */
050        public GroovyCompassQueryBuilder(CompassQueryBuilder queryBuilder) {
051            this.queryBuilder = queryBuilder;
052        }
053    
054        /**
055         * Build a CompassQuery for the given closure and return it
056         * @param closure a Closure which builds the query when executed
057         * @return a CompassQuery built by the given Closure
058         */
059        public CompassQuery buildQuery(Closure closure) {
060            CompassQueryBuildingClosureDelegate invoker = new CompassQueryBuildingClosureDelegate(queryBuilder);
061            closure = (Closure) closure.clone();
062            closure.setDelegate(invoker);
063            Object result;
064            if (closure.getMaximumNumberOfParameters() == 1) {
065                result = closure.call(queryBuilder);
066            } else {
067                result = closure.call();
068            }
069            CompassQuery query = invoker.getQuery();
070            return (CompassQuery) ((query == null) ? result : query);
071        }
072    
073        /**
074         * Directly invoke a builder method
075         */
076        public Object invokeMethod(String name, Object args) {
077            CompassQueryBuildingClosureDelegate invoker = new CompassQueryBuildingClosureDelegate(queryBuilder);
078            InvokerHelper.invokeMethod(invoker, name, args);
079            return invoker.getQuery();
080        }

083        /**
084         * This class acts as the query-builder closure delegate, identifying
085         * which method to call on which object and tying the result together at
086         * the end
087         *
088         * Note: Instances of this class are NOT thread-safe
089         */
090        private static class CompassQueryBuildingClosureDelegate extends GroovyObjectSupport {
091            private static final Log LOG = LogFactory.getLog(CompassQueryBuildingClosureDelegate.class);
092            private static final Map SHORT_METHOD_NAMES = shortMethodNames();
093            private static Map shortMethodNames() {
094                Map shortMethodNames = new HashMap();
095                shortMethodNames.put("should", "addShould");
096                shortMethodNames.put("must", "addMust");
097                shortMethodNames.put("mustNot", "addMustNot");
098                shortMethodNames.put("sort", "addSort");
099                return shortMethodNames;
100            }
101            private static final List BOOLEAN_ADDER_NAMES = booleanAdderNames();
102            private static List booleanAdderNames() {
103                List booleanAdderNames = new ArrayList();
104                booleanAdderNames.add("addShould");
105                booleanAdderNames.add("addMust");
106                booleanAdderNames.add("addMustNot");
107                return booleanAdderNames;
108            }
109            private static final Map SHORT_OPTION_NAMES = shortOptionNames();
110            private static Map shortOptionNames() {
111                Map shortOptionNames = new HashMap();
112                shortOptionNames.put("defaultProperty", "defaultSearchProperty");
113                shortOptionNames.put("andDefaultOperator", "useAndDefaultOperator");
114                shortOptionNames.put("parser", "queryParser");
115                return shortOptionNames;
116            }

118            private CompassQueryBuilder queryBuilder;
119            private Stack stack = new Stack();
120            private int depth = 0;
121            private Object previous; // previous method invocation result, if any
122    
123            public CompassQueryBuildingClosureDelegate(CompassQueryBuilder queryBuilder) {
124                this.queryBuilder = queryBuilder;
125            }
126    
127            public Object invokeMethod(String name, Object args) {
128                if (isTraceEnabled()) {
129                    trace("invokeMethod(" + name + ", " + args + ")");
130                }
131                depth++;
132    
133                // Remove Closure and options Map
134                Assert.isInstanceOf(Object[].class, args);
135                List invokeArgs = new ArrayList();
136                invokeArgs.addAll(Arrays.asList(((Object[]) args)));
137                Closure closure = (Closure) remove(invokeArgs, Closure.class);
138                Map options = (Map) remove(invokeArgs, Map.class);
139    
140                // Escape String queries?
141                if (name.equals("queryString") && options != null && Boolean.TRUE.equals(MapUtils.getBoolean(options, "escape"))) {
142                    Assert.isInstanceOf(String.class, invokeArgs.get(0));
143                    invokeArgs.set(0, QueryParser.escape((String) invokeArgs.get(0)));
144                    options.remove("escape");
145                }
146    
147                // Transform method name?
148                if (SHORT_METHOD_NAMES.containsKey(name)) {
149                    name = (String) SHORT_METHOD_NAMES.get(name);
150                }
151    
152                // Method to call is *on* CompassQuery but we have builder?
153                boolean queryMethod = hasMethod(CompassQuery.class, name) && !hasMethod(queryBuilder, name);
154                if (queryMethod && !hasMethod(peek(), name) && hasToQuery(peek())) {
155                    maybeAddPreviousShould(); // Lazy boolean?
156                    Object temp = pop();
157                    if (isTraceEnabled()) {
158                        trace("converting " + getShortClassName(temp) + " to query for " + name);
159                    }
160                    temp = toQuery(temp);
161                    push(temp);
162                } else if (queryMethod && !hasMethod(previous, name) && hasToQuery(previous)) {
163                    if (isTraceEnabled()) {
164                        trace("converting " + getShortClassName(previous) + " to query for " + name);
165                    }
166                    previous = toQuery(previous);
167                }
168    
169                // Implicit boolean?
170                if (BOOLEAN_ADDER_NAMES.contains(name) && !isWithinBool()) {
171                    trace("Implicit boolean -- " + name + " called when the stack is " + stack + ", args " + invokeArgs);
172                    Object bool = queryBuilder.bool();
173                    push(bool);
174                }
175    
176                // Method to call takes CompassQuery but we have Builder?
177                if (BOOLEAN_ADDER_NAMES.contains(name)) { // && !isQuery(invokeArgs[0])) {
178                    if (invokeArgs.isEmpty()) {
179                        Assert.notNull(closure, "Attempt to call " + name + " without a query or closure argument");
180                        maybeAddPreviousShould();
181                        trace("executing nested boolean closure");
182                        Object temp = previous;
183                        previous = null;
184                        Object innerBool = InvokerHelper.invokeMethod(this, "bool", closure);
185                        invokeArgs = new ArrayList();
186                        invokeArgs.add(innerBool);
187                        closure = null;
188                        previous = temp;
189                        trace("done nested boolean closure");
190                    }
191                    if (invokeArgs.size() > 0 && hasToQuery(invokeArgs.get(0))) {
192                        CompassQuery query = toQuery(invokeArgs.get(0));
193                        invokeArgs = new ArrayList();
194                        invokeArgs.add(query);
195                    }
196                }
197    
198                // Convert any BigDecimals to floats: simple but does the job for now
199                for (int i = 0; i < invokeArgs.size(); i++) {
200                    if (invokeArgs.get(i) instanceof BigDecimal) {
201                        invokeArgs.set(i, new Float(((BigDecimal) invokeArgs.get(i)).floatValue()));
202                    }
203                }
204    
205                Object result = null;
206                if (hasMethod(peek(), name)) {
207                    if (isTraceEnabled()) {
208                        trace("invoking " + peek().getClass().getName() + "." + name + "(" + invokeArgs+ ")");
209                    }
210                    result = InvokerHelper.invokeMethod(pop(), name, invokeArgs.toArray());
211                    if (isTraceEnabled()) {
212                        trace("result is " + result.getClass().getName() + " " + result);
213                    }
214                    push(result);
215                } else if (hasMethod(queryBuilder, name)) {
216                    // Eager implicit boolean check for previous should clause
217                    if (previous != null && !isWithinBool()) {
218                        trace("implicit boolean spotted");
219                        CompassQueryBuilder.CompassBooleanQueryBuilder bool = queryBuilder.bool();
220                        bool.addShould(toQuery(previous));
221                        push(bool);
222                        previous = null; // not necessary?
223                    } else {
224                        // Lazy boolean?
225                        maybeAddPreviousShould();
226                    }
227                    if (isTraceEnabled()) {
228                        trace("invoking queryBuilder." + name + "(" + invokeArgs + ")");
229                    }
230                    result = InvokerHelper.invokeMethod(queryBuilder, name, invokeArgs.toArray());
231                    if (isTraceEnabled()) {
232                        trace("result is " + result.getClass().getName() + " " + result);
233                    }
234                } else if (hasMethod(previous, name)) {
235                    if (isTraceEnabled()) {
236                        trace("invoking " + getShortClassName(previous) + "." + name + "(" + invokeArgs+ ")");
237                    }
238                    result = InvokerHelper.invokeMethod(previous, name, invokeArgs.toArray());
239                    if (isTraceEnabled()) {
240                        trace("result is " + result.getClass().getName() + " " + result);
241                    }
242                } else {
243                    throw new UnsupportedOperationException(
244                        "No such method CompassQueryBuilder#" + name +
245                        (peek() == null ? "" : " or " + getShortClassName(peek()) + "#" + name) +
246                        (previous == null ? "" : " or " + getShortClassName(previous) + "#" + name) +
247                        ". (Arguments were " + invokeArgs + ", stack is " + stack + ", result is " + result + ")" + //, this.result is ${this.result}) " +
248                        ". Refer to the Compass API docs at http://www.opensymphony.com/compass/versions/1.1/api/org/compass/core/CompassQueryBuilder.html to see what methods are available"
249                    );
250                }
251    
252                // Recurse into nested closure?
253                if (closure != null) {
254                    trace("invoking the closure arg");
255                    push(result);
256                    Object temp = previous;
257                    previous = null;
258                    closure = (Closure) closure.clone();
259                    closure.setDelegate(this);
260                    depth++;
261                    closure.call();
262                    depth--;
263    
264                    // Complete semi-implicit boolean?
265                    maybeAddPreviousShould();
266    
267                    previous = temp;
268                    result = pop();
269                }
270    
271                // Apply builder/query options?
272                result = applyOptions(result, options);
273                previous = result;
274                depth--;
275    
276                if (isTraceEnabled()) {
277                    trace("after methods and closure, depth " + depth + ", stack " + stack + ", previous " + previous);
278                    trace("returning " + result);
279                }
280                return result;
281            }
282    
283            private void maybeAddPreviousShould() {
284                if (previous != null && isWithinBool() && !previous.equals(peek())) {
285                    trace("previous lazy boolean should clause spotted");
286                    InvokerHelper.invokeMethod(peek(), "addShould", toQuery(previous));
287                    previous = null; // not necessary?
288                }
289            }

291            private Object applyOptions(Object result, Map options) {
292                if (options == null || options.size() == 0) {
293                    return result;
294                }
295    
296                // Convert shorthand to longhand names
297                Map tmp = new HashMap();
298                for (Iterator iter = options.entrySet().iterator(); iter.hasNext(); ) {
299                    Map.Entry entry = (Map.Entry) iter.next();
300                    String optionName = (String) entry.getKey();
301                    if (SHORT_OPTION_NAMES.containsKey(optionName)) {
302                        tmp.put(SHORT_OPTION_NAMES.get(optionName), entry.getValue());
303                    } else {
304                        tmp.put(optionName, entry.getValue());
305                    }
306                }
307                options = tmp;
308    
309                // Convert any BigDecimals to floats: simple but does the job for now
310                for (Iterator iter = options.entrySet().iterator(); iter.hasNext(); ) {
311                    Map.Entry entry = (Map.Entry) iter.next();
312                    String optionName = (String) entry.getKey();
313                    Object value = entry.getValue();
314                    if (value instanceof BigDecimal) {
315                        options.put(optionName, new Float(((BigDecimal) value).floatValue()));
316                    }
317                }
318    
319                if (isTraceEnabled()) {
320                    trace("applying options " + options + " to " + result);
321                }
322                Map queryOptions = new HashMap();
323    
324                boolean hasToQuery = !isQuery(result) && hasToQuery(result);
325                for (Iterator iter = options.entrySet().iterator(); iter.hasNext(); ) {
326                    Map.Entry entry = (Map.Entry) iter.next();
327                    String optionName = (String) entry.getKey();
328                    Object optionValue = entry.getValue();
329                    // special case for this single no-parameter, non-setter string query builder method
330                    if (optionName.equals("useAndDefaultOperator")) {
331                        if (optionValue != null) {
332                            Assert.isTrue(result instanceof CompassQueryBuilder.CompassMultiPropertyQueryStringBuilder || result instanceof  CompassQueryBuilder.CompassQueryStringBuilder, "'useAndDefaultOperator' option provided when current query/builder is a " + getShortClassName(result) + ", but should be a Compass*QueryStringBuilder");
333                            if (optionValue.equals(Boolean.TRUE)) {
334                                InvokerHelper.invokeMethod(result, "useAndDefaultOperator", null);
335                            } else {
336                                InvokerHelper.invokeMethod(result, "useOrDefaultOperator", null);
337                            }
338                        }
339                        continue;
340                    }
341                    if (optionName.equals("defaultOperator")) {
342                        Assert.notNull(optionValue, "'defaultOperator' option value is null: it must be one of 'or' or 'and'");
343                        Assert.isInstanceOf(String.class, optionValue, "'defaultOperator' option value is must be a String but is: [" + optionValue.getClass().toString() + "]");
344                        if (((String)optionValue).equalsIgnoreCase("or")) {
345                            InvokerHelper.invokeMethod(result, "useOrDefaultOperator", null);
346                        } else if (((String)optionValue).equalsIgnoreCase("and")) {
347                            InvokerHelper.invokeMethod(result, "useAndDefaultOperator", null);
348                        } else {
349                            throw new IllegalArgumentException("'defaultOperator' option value is not valid: it must be one of 'or' or 'and' but was '" + optionValue + "'");
350                        }
351                        continue;
352                    }
353                    String methodName = "set" + optionName.substring(0, 1).toUpperCase() + optionName.substring(1);
354                    trace("method is -- " + methodName + "(" + optionValue + ")");
355                    if (hasMethod(result, methodName)) {
356                        if (isTraceEnabled()) {
357                            trace("invoking " + getShortClassName(result) + "." + methodName + "(" + optionValue + ")");
358                        }
359                        result = InvokerHelper.invokeMethod(result, methodName, optionValue);
360                    } else if (hasToQuery && hasMethod(CompassQuery.class, methodName)) {
361                        trace("added option to queryOptions, " + queryOptions);
362                        queryOptions.put(optionName, optionValue);
363                    } else {
364                        throw new UnsupportedOperationException(
365                            "No such method " + getShortClassName(result) + "#" + methodName + (!hasToQuery ? "" : " or CompassQuery#" + methodName) +
366                            ". (Arguments were " + optionValue + ") " +
367                            ". Refer to the Compass API docs at http://www.opensymphony.com/compass/versions/1.1/api/org/compass/core/CompassQueryBuilder.html to see what methods are available"
368                        );
369                    }
370                }
371    
372                if (!queryOptions.isEmpty()) {
373                    trace("setting query options");
374                    result = toQuery(result);
375                    for (Iterator iter = queryOptions.entrySet().iterator(); iter.hasNext(); ) {
376                        Map.Entry entry = (Map.Entry) iter.next();
377                        String optionName = (String) entry.getKey();
378                        Object optionValue = entry.getValue();
379                        String methodName = "set" + optionName.substring(0, 1).toUpperCase() + optionName.substring(1);
380                        if (isTraceEnabled()) {
381                            trace("method is -- " + methodName + "(" + optionValue + ")");
382                        }
383                        Assert.isTrue(hasMethod(result, methodName));
384                        if (isTraceEnabled()) {
385                            trace("invoking " + getShortClassName(result) + "#" + methodName + "(" + optionValue + ")");
386                        }
387                        result = InvokerHelper.invokeMethod(result, methodName, optionValue);
388                    }
389                }
390    
391                return result;
392            }

394            public CompassQuery getQuery() {
395                if (previous != null) {
396                    boolean completeBoolean = false;
397                    if (isWithinBool() && !previous.equals(peek())) {
398                        if (isTraceEnabled()) {
399                            trace("Within bool " + peek() + ", and previous is " + previous);
400                        }
401                        completeBoolean = true;
402                    }
403                    if (!isQuery(previous)) {
404                        trace("converting result to query for caller");
405                        previous = toQuery(previous);
406                    }
407                    if (completeBoolean) {
408                        trace("completing boolean");
409                        InvokerHelper.invokeMethod(peek(), "addShould", previous);
410                        previous = toQuery(pop());
411                    }
412                }
413                return (CompassQuery) previous;
414            }
415    
416            private boolean isWithinBool() {
417                return isBoolBuilder(peek());
418            }
419    
420            private void push(Object object) {
421                stack.push(object);
422                if (isTraceEnabled()) {
423                    trace("pushed " + getShortClassName(object) + " " + object + " to stack");
424                }
425            }
426    
427            private Object pop() {
428                if (isTraceEnabled()) {
429                    trace("popping " + (peek() == null ? "null" : (getShortClassName(peek()) + " " + peek())) + " from the stack");
430                }
431                return stack.empty() ? null : stack.pop();
432            }
433    
434            private Object peek() {
435                return stack.empty() ? null : stack.peek();
436            }
437    
438            private boolean hasMethod(Object thing, String name) {
439                if (thing == null) {
440                    return false;
441                }
442                Class clazz = (Class) (thing instanceof Class ? thing : thing.getClass());
443                Method[] methods = clazz.getMethods();
444                for (int i = 0; i < methods.length; i++) {
445                    Method method = methods[i];
446                    if (method.getName().equals(name)) {
447                        return true;
448                    }
449                }
450                return false;
451            }
452    
453            private boolean hasToQuery(Object object) {
454                return object instanceof CompassQueryBuilder.ToCompassQuery;
455            }
456    
457            private CompassQuery toQuery(Object object) {
458                if (isQuery(object)) return (CompassQuery) object;
459                return ((CompassQueryBuilder.ToCompassQuery) object).toQuery();
460            }
461    
462            private boolean isQuery(Object object) {
463                return object instanceof CompassQuery;
464            }
465    
466            private boolean isBoolBuilder(Object object) {
467                return object instanceof CompassQueryBuilder.CompassBooleanQueryBuilder;
468            }
469    
470            private String getShortClassName(Object thing) {
471                if (thing == null) {
472                    return "null";
473                }
474                Class clazz = (Class) (thing instanceof Class ? thing : thing.getClass());
475                return ClassUtils.getShortName(clazz);
476            }
477    
478            private Object remove(List args, Class clazz) {
479                if (args == null) {
480                    return null;
481                }
482                for (int i = 0; i < args.size(); i++) {
483                    if (clazz.isAssignableFrom(args.get(i).getClass())) {
484                        return args.remove(i);
485                    }
486                }
487                return null;
488            }

490            private boolean isTraceEnabled() {
491    //        true
492                return LOG.isTraceEnabled();
493            }
494    
495            private void trace(String message) {
496                StringBuffer buf = new StringBuffer(message.length() + depth * 2);
497                for (int i = 0; i < depth; i++) {
498                    buf.append("  ");
499                }
500                buf.append(message);
501                LOG.trace(buf);
502    //        println(buf)
503            }
