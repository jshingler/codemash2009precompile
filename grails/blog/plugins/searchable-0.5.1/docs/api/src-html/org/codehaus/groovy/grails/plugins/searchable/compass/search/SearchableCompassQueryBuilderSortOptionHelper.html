016    package org.codehaus.groovy.grails.plugins.searchable.compass.search;
017    
018    import org.compass.core.CompassQuery;
019    import org.compass.core.Compass;
020    import org.compass.core.CompassSession;
021    import org.springframework.util.Assert;
022    import org.codehaus.groovy.grails.commons.GrailsApplication;
023    
024    import java.util.Map;
025    import java.util.List;
026    import java.util.Arrays;
027    
028    /**
029     * Post-processes a query to add sort
030     *
031     * @author Maurice Nicholson
032     */
033    public class SearchableCompassQueryBuilderSortOptionHelper implements SearchableCompassQueryBuilderOptionsHelper {
034        public static final String DIRECTION = "direction";
035        public static final String ORDER = "order";
036        public static final List VALID_SORT_DIRECTION_VALUES = Arrays.asList(new String[] { "asc", "desc", "auto", "reverse" });
037    
038        public CompassQuery applyOptions(GrailsApplication grailsApplication, Compass compass, CompassSession compassSession, CompassQuery compassQuery, Map options) {
039            return addSort(compassQuery, options);
040        }
041    
042        public CompassQuery addSort(CompassQuery compassQuery, Map options) {
043            String sort = (String) options.get("sort");
044            if (sort == null) {
045                return compassQuery;
046            }
047            Object sortProperty = getSortProperty(sort);
048            CompassQuery.SortDirection direction = getSortDirection(sortProperty, options);
049    
050            if (sortProperty instanceof CompassQuery.SortImplicitType) {
051                compassQuery = compassQuery.addSort((CompassQuery.SortImplicitType) sortProperty, direction);
052            } else {
053                Assert.isInstanceOf(String.class, sortProperty, "Expected string");
054                compassQuery = compassQuery.addSort((String) sortProperty, direction);
055            }
056            return compassQuery;
057        }
058    
059        private Object getSortProperty(String sort) {
060            Assert.notNull(sort, "sort cannot be null");
061            if (sort.equals("SCORE")) {
062                return CompassQuery.SortImplicitType.SCORE;
063            }
064            return sort;
065        }
066    
067        /**
068         * Get the CompassQuery.SortDirection for the given property and optional order/direction Map entry
069         * @param property either CompassQuery.SortImplicitType.SCORE or a class property name (String)
070         * @param options a Map containg
071         * @return
072         */
073        public CompassQuery.SortDirection getSortDirection(Object property, Map options) {
074            Assert.notNull(property, "sort property cannot be null");
075            Assert.notNull(options, "options Map cannot be null");
076            if (!options.containsKey(ORDER) && !options.containsKey(DIRECTION)) {
077                return CompassQuery.SortDirection.AUTO;
078            }
079    
080            Assert.isTrue((options.containsKey(ORDER) && !options.containsKey(DIRECTION)) || (!options.containsKey(ORDER) && options.containsKey(DIRECTION)), "Either specify a sort '" + ORDER + "' or '" + DIRECTION + "' or neither but not both");
081            String value = (String) options.get(DIRECTION);
082            if (value == null) {
083                value = (String) options.get(ORDER);
084            }
085            Assert.isTrue(VALID_SORT_DIRECTION_VALUES.contains(value), "The sort order/direction '" + value + "' is not a valid value");
086            return property.equals(CompassQuery.SortImplicitType.SCORE) ?
087                    value.equals("asc") || value.equals("reverse") ? CompassQuery.SortDirection.REVERSE : CompassQuery.SortDirection.AUTO :
088                    value.equals("asc") || value.equals("auto") ? CompassQuery.SortDirection.AUTO : CompassQuery.SortDirection.REVERSE;
089        }
090    }
